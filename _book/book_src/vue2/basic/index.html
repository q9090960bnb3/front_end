    new Vue({
      el: '#root',

        lastName: '三'
      }, 
      computed:{
        fullName:{
          // get:当有人读取时，get就会被调用
          // 初次读取时被调用（有缓存），所依赖得值发生变化也会调用

            this.firstName = arr[0]
            this.lastName = arr[1]
          }
        }
      }
